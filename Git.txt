Dir="/storage/emulated/0/Documents/Apps"

options=("clone" "status" "commit" "Quit")
select opt in "${options[@]}"
do
  case $opt in
  "clone")
    cd $Dir 
    git clone git@github.com:Haya-Sixt/DC.git
    rm -r Archive/DC
    mv DC Archive/DC
  ;;
  "status")
    cd $Dir/DC
    git status
  ;;
  "commit")
    cd $Dir/DC
    git add .
    git commit --all -m "default" 
  ;;
  "Quit")
    break
  ;;
  *) echo "invalid option $REPLY";;
esac done
